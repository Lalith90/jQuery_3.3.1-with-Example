<html>

      <script type = "text/javascript"

         $(document).ready(function(){
            $(document).bind("click", function (e) {
               $(e.target).closest("li").toggleClass("highlight");
            });
         });
      </script>

         .highlight { color:red; background: yellow;}
      </style>
