<html>

      <script type = "text/javascript" language = "javascript">
         $(document).ready(function(){
            $("div.demo").scrollTop( 200 );
            
            $("div.demo").mousemove(function () {
               var top = $("div.demo").scrollTop();
               $("#result").html("top offset: <span>" + top + "</span>.");
            });
            
         });
      </script>
      
      <style>
         div.demo {background:#CCCCCC none repeat scroll 0 0; border:3px

            width:200px; height:100px; overflow:auto;}

            margin:5px; float:left; background-color:blue;}
